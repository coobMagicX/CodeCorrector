protected double getInitialDomain(double p) {
    double ret;
    double d = getDenominatorDegreesOfFreedom();
    if (d > 2.0) {
        // use mean
        ret = d / (d - 2.0);
    } else {
        throw new IllegalArgumentException("Degrees of freedom must be greater than 2 for this operation.");
    }
    return ret;
}
